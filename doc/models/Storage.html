'use strict'; window.App.Storage = Ember.Object.extend({ base: 'coretheloothound_', getValue: function(key) { if (Modernizr.localstorage) { return window.localStorage.getItem(this.get('base') + key); } else { return Ember.$.cookie(this.get('base') + key); } }, setValue: function(key, value) { if (Modernizr.localstorage) { window.localStorage.setItem(this.get('base') + key, value); } else { Ember.$.cookie(this.get('base') + key, value); } }, removeValue: function(key) { if (Modernizr.localstorage) { window.localStorage.removeItem(this.get('base') + key); } else { Ember.$.removeCookie(this.get('base') + key); } } });
